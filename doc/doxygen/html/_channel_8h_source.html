00114     template<typename T> void Channel::addSampleRaw( const T rawval )
00115     {
00116         using boost::numeric_cast;
00117 
00118         switch( m_signalheader->get_datatype( ) )
00119         {
00120         case INT8: m_data->addSample( numeric_cast<int8>(rawval) ); break;
00121         case UINT8: m_data->addSample( numeric_cast<uint8>(rawval) ); break;
00122         case INT16: m_data->addSample( numeric_cast<int16>(rawval) ); break;
00123         case UINT16: m_data->addSample( numeric_cast<uint16>(rawval) ); break;
00124         case INT32: m_data->addSample( numeric_cast<int32>(rawval) ); break;
00125         case UINT32: m_data->addSample( numeric_cast<uint32>(rawval) ); break;
00126         case INT64: m_data->addSample( numeric_cast<int64>(rawval) ); break;
00127         case UINT64: m_data->addSample( numeric_cast<uint64>(rawval) ); break;
00128         case FLOAT32: m_data->addSample( numeric_cast<float32>(rawval) ); break;
00129         case FLOAT64: m_data->addSample( numeric_cast<float64>(rawval) ); break;
00130         default: throw exception::invalid_type_id( boost::lexical_cast<std::string>(m_signalheader->get_datatype( )) ); break;
00131         };
00132     }
